# NAME
cutadapt_folder.pl
# VERSION
0.9
# DEPENDENCIES
- `cutadapt`
# DESCRIPTION
Calls cutadapt on a folder and trims all files inside. After that the files are saved into another folder.
# PARAMETERS
- minlen = 17
- maxlen = 40
# INPUT
- In_folder
- out_folder
# OUTPUT
Trimmed reads with lengths between 17 and 40.
